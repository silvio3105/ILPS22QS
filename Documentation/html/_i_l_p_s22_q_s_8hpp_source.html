366 Return_t init(const interface_cfg_t* interfaceCfg = nullptr)
367 {
368 // Call MSP init handler if provided
369 if (mspInitHandler)
370 {
371 if (mspInitHandler() != Return_t::OK)
372 {
373 return Return_t::NOK;
374 }
375 }
376
377 // Set interface config
378 if (interfaceCfg)
379 {
380 if (interfaceConfig(interfaceCfg) != Return_t::OK)
381 {
382 return Return_t::NOK;
383 }
384 }
385
386 // Check sensor ID
387 uint8_t tmp = 0;
388 if (whoAmI(tmp) == Return_t::OK)
389 {
390 if (tmp != chipID)
391 {
392 return Return_t::NOK;
393 }
394 }
395 else
396 {
397 return Return_t::NOK;
398 }
399
400 // Cache pressure scale
401 return readPressureScale();
402 }

410 Return_t deinit(void) const
411 {
412 // Call deinit handler if provided
413 if (mspDeinitHandler)
414 {
415 return mspDeinitHandler();
416 }
417
418 return Return_t::OK;
419 }

429 Return_t setInterruptConfig(const interrupt_cfg_t& config)
430 {
431 txBuffer[0] = (uint8_t)Register_t::Interrupt;
432 txBuffer[1] = ((uint8_t)config.autoREFP << (uint8_t)InterruptBitmap_t::AutoREFP) |
433 ((uint8_t)config.autoZero << (uint8_t)InterruptBitmap_t::AutoZero) |
434 ((uint8_t)config.interruptLatch << (uint8_t)InterruptBitmap_t::InterruptLatch) |
435 ((uint8_t)config.pressureHighInterrupt << (uint8_t)InterruptBitmap_t::PressureHighEvent) |
436 ((uint8_t)config.pressureLowInterrupt << (uint8_t)InterruptBitmap_t::PressureLowEvent) |
437 ((uint8_t)config.resetARP << (uint8_t)InterruptBitmap_t::ResetAutoREFP) |
438 ((uint8_t)config.resetAZ << (uint8_t)InterruptBitmap_t::ResetAutoZero);
439
440 return writeRegister(txBuffer, 2);
441 }

463 config.pressureLowInterrupt = (State_t)((tmp >> (uint8_t)InterruptBitmap_t::PressureLowEvent) & 1);

466 return Return_t::OK;

477 Return_t setPressureInterruptThreshold(const uint16_t threshold)
478 {
479 // Calculate raw pressure value and set TX buffer
480 const uint16_t rawPressure = threshold * pressureScaleDivider[(uint8_t)getPressureScale()];
481 txBuffer[0] = (uint8_t)Register_t::PressureThresholdHigh;
482 txBuffer[1] = rawPressure >> 8;
483
484 if (writeRegister(txBuffer, 2) != Return_t::OK)
485 {
486 return Return_t::NOK;
487 }
488
489 txBuffer[0] = (uint8_t)Register_t::PressureThresholdLow;
490 txBuffer[1] = rawPressure & 0xFF;
491
492 if (writeRegister(txBuffer, 2) != Return_t::OK)
493 {
494 return Return_t::NOK;
495 }
496
497 return Return_t::OK;
498 }

508 Return_t getPressureInterruptThreshold(uint16_t& output)
509 {
510 uint16_t tmpOutput = 0;
511 uint8_t tmp = 0;
512
513 // Read high byte
514 if (readRegister(Register_t::PressureThresholdHigh, tmp) != Return_t::OK)
515 {
516 return Return_t::NOK;
517 }
518 tmpOutput = tmp << 8;
519
520 // Read low byte
521 if (readRegister(Register_t::PressureThresholdLow, tmp) != Return_t::OK)
522 {
523 return Return_t::NOK;
524 }
525 tmpOutput |= tmp;
526
527 // Convert to output unit
528 output = tmpOutput / pressureScaleDivider[(uint8_t)getPressureScale()];
529 return Return_t::OK;
530 }

540 Return_t setPressureScale(const PressureScale_t scale)
541 {
542 // Read control 2 register
543 uint8_t tmp = 0;
544
545 if (readRegister(Register_t::Control2, tmp) != Return_t::OK)
546 {
547 return Return_t::NOK;
548 }
549
550 txBuffer[0] = (uint8_t)Register_t::Control2;
551 txBuffer[1] = tmp & ~(1 << (uint8_t)Control2Bitmap_t::FullScale);
552 txBuffer[1] |= ((uint8_t)scale << (uint8_t)Control2Bitmap_t::FullScale);
553
554 if (writeRegister(txBuffer, 2) != Return_t::OK)
555 {
556 return Return_t::NOK;
557 }
558
559 pressureScale = scale;
560 return Return_t::OK;
561 }
