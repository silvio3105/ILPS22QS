404 Return_t init(const interface_cfg_t* interfaceCfg = nullptr)
405 {
406 ILPS22QS_DEBUG_INFO_PRINT("Version: ", 9);
407 ILPS22QS_DEBUG_INFO_PRINT(version, sizeof(version) - 1);
408 ILPS22QS_DEBUG_INFO_PRINT("\n", 1);
409
410 // Call MSP init handler if provided
411 if (mspInitHandler)
412 {
413 if (mspInitHandler() != Return_t::OK)
414 {
415 ILPS22QS_DEBUG_ERROR_PRINT("MSP init fail\n", 14);
416 return Return_t::NOK;
417 }
418 }
419
420 // Set interface config
421 if (interfaceCfg)
422 {
423 if (interfaceConfig(interfaceCfg) != Return_t::OK)
424 {
425 ILPS22QS_DEBUG_ERROR_PRINT("Interface fail\n", 15);
426 return Return_t::NOK;
427 }
428 }
429
430 // Check sensor ID
431 uint8_t tmp = 0;
432 if (whoAmI(tmp) == Return_t::OK)
433 {
434 if (tmp != chipID)
435 {
436 ILPS22QS_DEBUG_ERROR_PRINT("Invalid chip ID\n", 16);
437 return Return_t::NOK;
438 }
439 }
440 else
441 {
442 ILPS22QS_DEBUG_ERROR_PRINT("WhoAmI fail\n", 12);
443 return Return_t::NOK;
444 }
445
446 // Cache pressure scale
447 return readPressureScale();
448 }

525 Return_t setPressureInterruptThreshold(const uint16_t threshold)
526 {
527 // Calculate raw pressure value and set TX buffer
528 const uint16_t rawPressure = threshold * pressureScaleDivider[(uint8_t)getPressureScale()];
529 txBuffer[0] = (uint8_t)Register_t::PressureThresholdHigh;
530 txBuffer[1] = rawPressure >> 8;
531
532 if (writeRegister(txBuffer, 2) != Return_t::OK)
533 {
534 return Return_t::NOK;
535 }
536
537 txBuffer[0] = (uint8_t)Register_t::PressureThresholdLow;
538 txBuffer[1] = rawPressure & 0xFF;
539
540 if (writeRegister(txBuffer, 2) != Return_t::OK)
541 {
542 return Return_t::NOK;
543 }
544
545 ILPS22QS_DEBUG_INFO_PRINTF("Interrupt threshold set to %u\n", threshold);
546 return Return_t::OK;
547 }

557 Return_t getPressureInterruptThreshold(uint16_t& output)
558 {
559 uint16_t tmpOutput = 0;
560 uint8_t tmp = 0;
561
562 // Read high byte
563 if (readRegister(Register_t::PressureThresholdHigh, tmp) != Return_t::OK)
564 {
565 return Return_t::NOK;
566 }
567 tmpOutput = tmp << 8;
568
569 // Read low byte
570 if (readRegister(Register_t::PressureThresholdLow, tmp) != Return_t::OK)
571 {
572 return Return_t::NOK;
573 }
574 tmpOutput |= tmp;
575
576 // Convert to output unit
577 output = tmpOutput / pressureScaleDivider[(uint8_t)getPressureScale()];
578 return Return_t::OK;
579 }

589 Return_t setPressureScale(const PressureScale_t scale)
590 {
591 // Read control 2 register
592 uint8_t tmp = 0;
593
594 if (readRegister(Register_t::Control2, tmp) != Return_t::OK)
595 {
596 return Return_t::NOK;
597 }
598
599 txBuffer[0] = (uint8_t)Register_t::Control2;
600 txBuffer[1] = tmp & ~(1 << (uint8_t)Control2Bitmap_t::FullScale);
601 txBuffer[1] |= ((uint8_t)scale << (uint8_t)Control2Bitmap_t::FullScale);
602
603 if (writeRegister(txBuffer, 2) != Return_t::OK)
604 {
605 return Return_t::NOK;
606 }
607
608 pressureScale = scale;
609 ILPS22QS_DEBUG_INFO_PRINTF("Pressure scale set to %02X\n", txBuffer[1]);
610 return Return_t::OK;
611 }

1031
1032 if (readRegister(Register_t::PressureOutHigh, reg) != Return_t::OK)

1048 tmp |= reg;
